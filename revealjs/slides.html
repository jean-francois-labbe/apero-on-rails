Fonctions

def welcome
  puts "Welcome here"
end

welcome # => Welcome here

welcome() # => Welcome here

def welcome(name)
  puts "Welcome #{name}"
end

welcome "bob" # => Welcome bob

welcome("bob") # => Welcome bob

Keyword arguments

def welcome(name: "bob")
  puts "Welcome #{name}"
end

welcome # => Welcome bob

welcome("john") # => Welcome john

Class

class World
  def initialize(name)
    @name = name # Instance variable
  end

  def name
    @name
  end

  def name=(name)
    @name = name
  end
end
world = World.new("level 1")
world.name # => "level 1"
world.name = "new world" # => "new world"
